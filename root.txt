#include <stdio.h>
#include <stdlib.h>
typedef struct node
{
int data;
struct node *next;
} NODE;
void append(NODE **head, int n)
{
NODE *newnode;
newnode = malloc(sizeof(NODE));
newnode->data = n;
if (*head == NULL)
newnode->next = newnode;
else
{
newnode->next = (*head)->next;
(*head)->next = newnode;
}
*head = newnode;
}
